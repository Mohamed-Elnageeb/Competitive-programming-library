def hash_collision(n):
    mask = (1<<17) - 1
    fill = int((1<<15)*1.3+1)
    arr = [mask+2]*2
    x = 6
    for i in range(1,fill):
        arr += [x]+[x]
        x = x * 5 + 1
        x = x & mask

    arr += [1]*(n-len(arr))
    return arr
n = 2*10**5
x = 10**9
y = 10**9
